#include<iostream>
#include<string>
#include<cstdio>
#include<algorithm>
using namespace std;
void is(int n)
{
	if(n==1)
	cout<<"YES";
	else
	cout<<"NO";
}
int main()
{
	int n;
	scanf("%d",&n);
	string str;
	int aa[n][3];
	
	fill(aa[0],aa[0]+n*3,0);
	
	int out[n]={0};
	for(int i=0;i<n;i++)
	{
		cin>>str;
		
		for(int j=0;j<str.size() ;j++)
		{
			if(str[j]=='P')
			aa[i][0]++;
			else if(str[j]=='A')
			aa[i][1]++;
			else if(str[j]=='T')
			aa[i][2]++;
			else
			aa[i][0]=-100;
		}
		str.clear() ;
	}
	
	for(int i=0;i<n;i++)
	{
		if((aa[i][0]>0)&&(aa[i][1]>0)&&(aa[i][2]>0))
		out[i]=1;
		else
		out[i]=0;
	}
	
	for(int i=0;i<n;i++)
	{
		if(i!=(n-1))
		{
			is(out[i]);
			cout<<endl;
		}
		else
			is(out[i]);
	}
	
	
	return 0;
 }
