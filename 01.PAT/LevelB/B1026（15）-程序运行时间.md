	int sum=(int)round((double)(b-a)/100);

	int three=s%60;//秒钟 
	
	int two=(s/60);//fen

		printf("%d:",one);
	}

	{
		printf("0%d:",one);
	}
	
	if(two>=10)
	{
		printf("%d:",two);
	}
	else
	{
		printf("0%d:",two);
	}

	}
		
	return 0;	
}
