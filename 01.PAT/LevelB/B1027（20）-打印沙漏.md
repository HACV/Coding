```cpp
#include<bits/stdc++.h>
using namespace std;



int solution(int n)
{
	int num=25;
	
	for(;num>0;--num)
	{
		if(2*num*num<=n)
		{
			break;
		}
	}
	
	return num;
}

void test(int num,char c)
{
	//uploop
	int up_loop=num;
	int up_num=2*num-1;
	
	//down loop
	int down_loop=num-1;
	int down_num=3;
	
	//前面空格的个数 
	int count=0;
	for( ;up_loop>0 ;  )
	{
		for(int i=0;i<count;++i)
		{
			printf(" ");
		}
		
		
		
		for(int i=0;i<up_num;++i)
		{
			printf("%c",c);
		}
		printf("\n");
		
		//下次空格 
		++count;
		//下次符号 
		up_num-=2;
		//还剩下的循环 
		--up_loop; 
	}
	
	count-=2;
	
	for( ;down_loop>0 ;  )
	{
		for(int i=0;i<count;++i)
		{
			printf(" ");
		}
		
		
		
		for(int i=0;i<down_num;++i)
		{
			printf("%c",c);
		}
		printf("\n");
		
		//下次空格 
		--count;
		//下次符号 
		down_num+=2;
		//还剩下的循环 
		--down_loop; 
	}
	
}


int main()
{
	int n;
	char c;
	while(~scanf("%d %c",&n,&c))
	{
		int num=solution(n);
		if(0!=num) 
		{
			test(num,c);
			printf("%d\n",n-(2*num*num-1));
		}
		else
		{
			//表示n=1
			printf("%c\n",c); 
			printf("0\n");
		}
		
		
	}
	
	return 0;
}
```
