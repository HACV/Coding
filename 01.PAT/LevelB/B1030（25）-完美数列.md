## 双指针
牛客网上AC了，但PAT上显示有一个答案错误
```cpp
#include<bits/stdc++.h>
using namespace std;
const int maxn=100000+1;
int test[maxn]={0};
int n,p;


int main()
{
	
	while(~scanf("%d%d",&n,&p))
	{
		for(int i=0;i<n;++i)
		{
			scanf("%d",&test[i]);
		}
		sort(test,test+n);
		
		int solution=1;
		//双指针法 
		int i=0,j=0;
		while((i<n)&&(j<n))
		{
			while((j<n)&&(test[j]<=(long long)(test[i]*p)))
			{
				solution=max(solution,j-i+1);
				++j;
			}
			++i;
		}
		printf("%d\n",solution);
	}
	
	
	return 0;
}
```
