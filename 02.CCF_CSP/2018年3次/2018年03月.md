## 一、跳一跳

```cpp
#include<bits/stdc++.h>
using namespace std;


int solve[35];
int len=0;
int sum=0;

//除去第一次之外的选择返回
int select(int prev, int selectNum )
{
    if( 1==selectNum )
    {
        return 1;
    }
    else
    {
        if( 1==prev )
        {
            return 2;
        }
        else
        {
            return prev+2;
        }
    }
}


int main()
{

    int temp;
    len=0;

    while( ~scanf("%d",&temp) )
    {
        if( 0==temp )
        {
            break;
        }
        else
        {
            solve[len++]=temp;
        }
    }
    

    sum=0;
    int prev=1;
    for(int i=0; i<len; ++i)
    {
        if( 0==i )
        {
            if( 1==solve[i] )
            {
                sum+=1;
                prev=1;
            }
            else
            {
                sum+=2;
                prev=2;
            }
            
        }
        else
        {
            sum+=select( prev, solve[i] );
            prev=select( prev, solve[i] );
        }

    }

    printf("%d\n",sum);
    

    return 0;
}   
```
