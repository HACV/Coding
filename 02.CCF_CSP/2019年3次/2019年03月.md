## 一、小中大


考点：模拟


```cpp
#include<bits/stdc++.h>
using namespace std;

const int maxn=100000+5;
int solution[maxn];

int main()
{
	int n;
	while(~scanf("%d",&n))
	{
		for(int i=0;i<n;++i)
		{
			scanf("%d",&solution[i]);
		}
		
		int _max,_min;
		sort(solution,solution+n);
		_max=solution[n-1];
		_min=solution[0];
		
		//奇数 
		if(n&1)
		{
			printf("%d %d %d\n",_max,solution[n/2],_min);
		}
		else
		{
			int mid_two=solution[n/2-1]+solution[n/2];
			
			if(0==mid_two%2)//能整除 
			{
				printf("%d %d %d\n",_max,mid_two/2,_min);
			}
			else
			{
				//由于都是整数，所以，不需要考虑四舍五入，小数点必定是.5 
				printf("%d %.1f %d\n",_max,(double)mid_two/(double)2,_min);
			}
		}
	}
	
	
	
	return 0;
} 
```


## 二、二十四点

考点：栈和队列
乍一看，还以为有括号啥的，以为和PAT那题一样，以为要先弄成中缀表达式。
但是仔细看题，很多条件都简化了，所以直接用裸的“栈”和“队列”来解决
但是我打算用双端队列来解决。
代码不好看，其实主要是像复习STL中双端队列的用法。


```cpp
#include<bits/stdc++.h>
using namespace std;

char solution[15];

int main()
{
	//优先级设置 
	map<char,int> mp;
	mp['+']=1;
	mp['-']=2;
	mp['x']=3;
	mp['/']=4;
	
	
	int n;
	while(~scanf("%d",&n))
	{
		for(int i=0;i<n;++i)
		{
			scanf("%s",solution);
			deque<int> num;
			num.push_back(solution[0]-'0');
			
			queue<char> myOperator;
			
			//搞定优先级3-4 
			for(int j=1;j<=5;j+=2)
			{
				if(mp[solution[j]]>2)
				{
					int a=num[num.size()-1];
					num.pop_back();
					
					int b=solution[j+1]-'0';
					if('x'==solution[j])
					{
						num.push_back(a*b);
					}
					else
					{
						num.push_back(a/b);
					}
					
				}
				else
				{
					num.push_back(solution[j+1]-'0');
					myOperator.push(solution[j]);
				}
				
			}
			
			
			//搞定优先级1-2 
			while(!myOperator.empty())
			{
				if('+'==myOperator.front())
				{
					int a=num[0];
					int b=num[1];
					num.pop_front();
					num.pop_front();
					num.push_front(a+b);
				}
				else
				{
					int a=num[0];
					int b=num[1];
					num.pop_front();
					num.pop_front();
					num.push_front(a-b);
				}
				
				myOperator.pop();
			}
			
			
			if(24==num[0])
			{
				printf("Yes\n");
			}
			else
			{
				printf("No\n");
			}
			
			//还有个坑，记得把num清空
			num.pop_back(); 
		}
	}
	
	
	
	return 0;
} 
```
