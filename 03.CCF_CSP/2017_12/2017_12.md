## 1）最小差值（签到题）
考点：模拟

下面测试显示，判卷系统底层是`Linux`，可以用万能头文件。


```cpp
#include<bits/stdc++.h>
using namespace std;

int test[1005];

int main()
{
	int n;
	while(~scanf("%d",&n))
	{
		for(int i=0;i<n;++i)
		{
			scanf("%d",&test[i]);
		}
		
		sort(test,test+n);
		int _min=999999;
		for(int i=1;i<n;++i)
		{
			if(test[i]-test[i-1]<_min)
			{
				_min=test[i]-test[i-1];
			}
		}
		
		printf("%d\n",_min);
	}
	
	return 0;//注意，一定要return 0，否则会运行错误是0分
}
```
