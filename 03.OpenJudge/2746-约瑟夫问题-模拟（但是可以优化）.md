## 方法一，直接模拟
思想来源于，数据结构一书中，学循环队列的时候，取模

```cpp
#include<stdio.h>
#include<string.h>

int main()
{
	int array[301]={0};//表示数字在
	
	int n,m;
	
	while((~scanf("%d%d",&n,&m))&&(0!=n)&&(0!=m))
	{
		memset(array,0,sizeof(array));
		int count=0;//表示数数字到几号了 
		int sum=0;
		int i=1;//当前序号 
		while(1)
		{
			
			
			//判断有没有被筛 
			if(0==array[i])//没有被筛掉了 
			{
				++count; //数数字 
				
			}
			
			//判断需不需要筛掉
			if(m==count)
			{
				array[i]=1;
				++sum;//筛掉了一个 
				count=0;//继续下一次 
				
			}
			
			if(sum==(n-1)) 
			{
				
				break;
			}
			
			++i;
			
			if((n+1)==i)
			{
				i=1;			
			}
		}
		
		
		for(i=1;i<=n;++i)
		{
			if(!array[i])
			{
				printf("%d\n",i);
			}
		}
		
	} 
	
	
	return 0;
}
```
