#include<cstdio>
#include<cstring>

int test[10000+1]={0};

int main()
{
	int L,M;
	
	while(~scanf("%d%d",&L,&M))
	{
		memset(test,0,sizeof(test));
		for(int i=0;i<M;++i)
		{
			int a,b;
			scanf("%d%d",&a,&b);
			while(a<=b)
			{
				test[a]=1;//标记要挖 
				++a;
			}
		}
		
		int sum=0;
		for(int i=0;i<=L;++i)
		{
			if(0==test[i])
			{
				++sum;
			}
		}
		
		printf("%d\n",sum);
	}
	
	
	return 0;
}
